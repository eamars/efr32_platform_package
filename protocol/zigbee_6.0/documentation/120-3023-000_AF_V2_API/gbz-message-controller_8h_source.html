237 /*
238  * @brief This function initializes proper states for construction of GBZ messages.
239  *
240  * Depending on the value of the argument (gbzCommand), the creator will
241  * behave differently. If a NULL value is passed, the creator assumes the
242  * user wants the API to allocate memory to store the appended responses.
243  * Otherwise, the creator will use the provided buffer as the destination for
244  * storing responses.
245  *
246  * Below is a general flow for the creation of ZCL messages into a GBZ
247  * message.
248  *
249  * 1. emberAfPluginGbzMessageControllerCreatorInit() - create
250  * 2. emberAfPluginGbzMessageControllerAppendCommand() - append
251  * 3. emberAfPluginGbzMessageControllerCreatorAssemble() - assemble result.
252  * 4. emberAfPluginGbzMessageControllerCreatorCleanup() - memory clean up
253  *
254  * @param state A pre-allocated struct that will be updated to hold
255  * bookkeping information for creating GBZ messages
256  * @param type GBZ payload type: command, response, or alert
257  * @param alertCode When type is alert this field contains the alert code
258  * @param timestamp When type is alert this field contains the UTC when the alert occurred
259  * @param gbzCommand NULL - if the user wants API to allocate memory to store
260  * respones
261  * Otherwsie - pointer to destination buffer for gbz messages
262  * @param gbzCommandLength length of the gbz messages. this argument is
263  * ignored if gbzCommand is NULL
264  *
265  */
266 uint16_t emberAfPluginGbzMessageControllerCreatorInit(EmberAfGbzMessageCreatorState * state,

268  uint16_t alertCode,
269  uint32_t timestamp,
