89 static void emberAfPluginReportingSetDefaultReportingConfiguration(EmberAfPluginReportingEntry *defaultConfiguration, int *count)
90 {
91  int size = sizeof(mandatoryReportableTable) / sizeof(EmberAfPluginReportingMandatoryReportableAttributeEntry);
92 
93  for (int i = 0; i < size; i++) {
94  // Find if the endpoint has a cluster that can have mandatory attributes
95  if (emberAfContainsCluster(defaultConfiguration->endpoint, mandatoryReportableTable[i].clusterId)) {
96  defaultConfiguration->clusterId = mandatoryReportableTable[i].clusterId;
97  // Find if the cluster has the mandatory reportable attribute
98  if ( emberAfContainsServer(defaultConfiguration->endpoint, defaultConfiguration->clusterId)) {
99  defaultConfiguration->attributeId = mandatoryReportableTable[i].attributeId;
100  defaultConfiguration->data.reported.minInterval = mandatoryReportableTable[i].minInterval;
101  defaultConfiguration->data.reported.maxInterval = mandatoryReportableTable[i].maxInterval;
102  defaultConfiguration->data.reported.reportableChange = mandatoryReportableTable[i].reportableChange;
103  if ((EMBER_AF_PLUGIN_REPORTING_TABLE_SIZE) > *count ) {
104  MEMMOVE(&generatedReportingConfigDefaults[*count],
105  defaultConfiguration,
106  sizeof(EmberAfPluginReportingEntry));
107  *count = (*count) + 1;
108  } else {
109  return;
110  }
111  }
112  }
113  }
114 }

116 static void emberAfPluginReportingInitSetupDefaultConfiguration(int *count)
117 {
118  // set up common reporting entry parameters here.
119  EmberAfPluginReportingEntry defaultConfiguration = {
120  EMBER_ZCL_REPORTING_DIRECTION_REPORTED, //direction

125  EMBER_AF_NULL_MANUFACTURER_CODE, //manufacturerCode
126  .data.reported = {
127  1, //minInterval
128  0xffff, //maxInterval
129  1 //reportableChange
130  }
131  };
132 
133  for (int i = 0; i < emberAfEndpointCount(); i++) {
134  defaultConfiguration.endpoint = emberAfEndpointFromIndex(i);
135  emberAfPluginReportingSetDefaultReportingConfiguration(&defaultConfiguration,
136  count);
137  }
