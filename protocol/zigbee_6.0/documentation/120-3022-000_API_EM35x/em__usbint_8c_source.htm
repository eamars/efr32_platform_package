251 void VBUSMON_ISR(void)
252 {
253  //VBUSMON is used for sensing VBUS indicating if USB is physically connected.
254  //An IRQ is used to monitor for any edge change on VBUSMON.
255  //For self-powered devices, VBUS monitoring is required for the EM358 device
256  //to function cleanly across physical connect/disconnects of the USB.
257  //The USB specification requires that the pull-up resistor for
258  //enumerating is disconnected if VBUS is not connected.
259 
260  //clear int before read to avoid potential of missing interrupt
261  EVENT_MISS->MISS = VBUSMON_MISS_BIT; //clear missed interrupt flag
262  EVENT_GPIO->FLAG = VBUSMON_FLAG_BIT; //clear top level interrupt flag
263 
264  //Because this ISR/IRQ triggers on both edges, usbVbusPresent is used
265  //to track the state of VBUS.
266  //Always directly read VBUSMON to ensure this ISR is changing to the correct,
267  //current state.
268  //Falling edge: VBUS removed - Must set ENUMCTRL to input-low.
269  //Rrising edge: VBUS applied - If usbConfigEnumerate() has been called and
270  //this device is ready to enumerate, set ENUMCTRL to output-high and
271  //perform enumeration.
272  usbVbusPresent = ((VBUSMON_IN & VBUSMON) == VBUSMON);
273 
274  if (usbVbusPresent) {
275  if (USBD_GetUsbState() == USBD_STATE_CONFIGURED) {
276  return;
277  }
278  //Only attempt enumerate if usbConfigEnumerate has already been called.
279  //Set ENUMCTRL ouput-high to enumerate.
280  ENUMCTRL_SETCFG(_GPIO_P_CFGL_Px0_OUT);
281  ENUMCTRL_SET();
282 
283  // halSetLed(BOARDLED0);
284  USBD_SetUsbState(USBD_STATE_DEFAULT);
285  } else {
286  //Configure ENUMCTRL as an input so the device de-enumurates and
287  //the pin does not put any load on the bus. (The spec says the
288  //pull-up resistor used for enumeration should be tri-state. On
289  //EM358 devices the input configuration is the best available choice
290  //since tri-state isn't possible.)
291  //Clear down ENUMCTRL to be ready for the next time enumerating.
292  ENUMCTRL_SETCFG(_GPIO_P_CFGL_Px0_IN);
293  ENUMCTRL_CLR();
294  // halClearLed(BOARDLED0);
295  USBD_SetUsbState(USBD_STATE_NONE);
296  }
297 }
